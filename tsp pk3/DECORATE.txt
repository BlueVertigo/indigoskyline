//Melee Button Press

Actor Boolean : Inventory
{
    Inventory.MaxAmount 1
    +UNDROPPABLE
    -INVBAR
}

Actor MeleePressed : Boolean {}
Actor DropShellsActor : Boolean {}

Actor StartMelee : CustomInventory
{
  -INVBAR
  +UNDROPPABLE
  Inventory.Amount 1
  Inventory.MaxAmount 1
  States
  {
      Use:
      TNT1 A 0 A_GiveInventory("MeleePressed", 1)
	  TNT1 A 0 A_GiveInventory("MakeHellFireBurst", 1)
      Fail
  }
}

Actor StopMelee : CustomInventory
{
  -INVBAR
  +UNDROPPABLE
  Inventory.Amount 1
  Inventory.MaxAmount 1
  States
  {
      Use:
      TNT1 A 0 A_TakeInventory("MeleePressed", 1)
	  TNT1 A 0 A_TakeInventory("MakeHellFireBurst", 1)
      Fail
  }
}

Actor DropCurrentShells : CustomInventory
{
  -INVBAR
  +UNDROPPABLE
  Inventory.Amount 1
  Inventory.MaxAmount 1
  States
  {
      Use:
      TNT1 A 0 A_GiveInventory("DropShellsActor", 1)
      Fail
  }
}

ACTOR MeleeButtonActor : CustomInventory
{
  -INVBAR
  +UNDROPPABLE
  Inventory.Amount 1
  Inventory.MaxAmount 1
  States
  {
      Use:
	  TNT1 A 0 A_JumpIfInventory("MeleeSelected",1,"MeleeSwitch")
	  TNT1 A 0 A_SelectWeapon("MelMelee")
	  Fail
	  MeleeSwitch:
	  TNT1 A 0 A_JumpIfInventory("919Selected",1,"919Switch")
	  TNT1 A 0 A_JumpIfInventory("ZekeSelected",1,"ZekeSwitch")
	  TNT1 A 0 A_JumpIfInventory("DahliaSelected",1,"DahliaSwitch")
	  TNT1 A 0 A_JumpIfInventory("AmmoBoxSelected",1,"AmmoBoxSwitch")
	  TNT1 A 0 A_JumpIfInventory("ToolBoxSelected",1,"ToolBoxSwitch")
	  Goto 919Switch
	  919Switch:
	  TNT1 A 0 A_SelectWeapon("919mPistol")
	  Fail
	  ZekeSwitch:
	  TNT1 A 0 A_SelectWeapon("ZekeShotgun")
	  Fail
	  DahliaSwitch:
	  TNT1 A 0 A_SelectWeapon("DahliaCrossbow")
	  Fail
	  AmmoBoxSwitch:
	  TNT1 A 0 A_SelectWeapon("AmmoBox50")
	  Fail
	  ToolBoxSwitch:
	  TNT1 A 0 A_SelectWeapon("Toolbox")
	  Fail
  }
}
